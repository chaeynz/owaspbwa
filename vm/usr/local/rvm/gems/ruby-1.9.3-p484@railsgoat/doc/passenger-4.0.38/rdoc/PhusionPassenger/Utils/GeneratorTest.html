def test_hash
  json = generate(:abc => 123, 123 => 'abc')
  assert_match /^\{/, json
  assert_match /\}$/, json
  assert_equal [%Q("123": "abc"), %Q("abc": 123)], json[1...-1].split(', ').sort
end
